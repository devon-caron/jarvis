 Plan: Rework Model Loading System (Multi-GPU, Multi-Model)                                                 

 Context

 The current daemon loads models directly in-process using LlamaBackend + ModelManager. Because
 CUDA_VISIBLE_DEVICES must be set before CUDA initializes in a process, and only one model is supported at a
 time, independent GPU assignment per model is impossible in the current design. This rework replaces the
 monolithic daemon with a coordinator + worker subprocess pattern, where the main daemon becomes a pure
 coordinator and each loaded model runs as a separate subprocess with its own CUDA_VISIBLE_DEVICES
 environment variable.

 ---
 Architecture

 Client → Main Daemon (coordinator, no CUDA)
               ├── Registry: modelPath → WorkerInfo{PID, socket, GPUs, timeout}
               ├── GPU lock map: deviceID → modelPath
               └── Workers (one per loaded model):
                     _worker subprocess, CUDA_VISIBLE_DEVICES=0,1
                     owns a Unix socket (/tmp/jarvis-<uid>-worker-<hash>.sock)
                     loads one model via LlamaBackend+ModelManager+Server+Handler

 Chat requests are proxied via the main daemon: main daemon looks up the right worker and forwards the NDJSON
  request line, streaming response lines back to the client.

 ---
 New User-Facing Behavior

 # Load model on GPUs 0 and 1, auto-unload after 1 hour
 jarvis load -g 0,1 -t 3600 /path/to/model.gguf

 # Load another model on GPU 2, no timeout
 jarvis load -g 2 /path/to/other-model.gguf

 # Chat with a specific loaded model
 jarvis -m /path/to/model.gguf "hello"

 # Unload a specific model
 jarvis unload /path/to/model.gguf

 # Status shows all workers
 jarvis status

 Step 3 — Internal: Add worker socket path

 File: internal/paths.go

 import "hash/fnv"

 // WorkerSocketPath returns a deterministic socket path for a worker identified by id.
 // id should be a stable string like "modelpath|gpulist".
 func WorkerSocketPath(id string) string {
     h := fnv.New32a()
     h.Write([]byte(id))
     hash := fmt.Sprintf("%08x", h.Sum32())
     if dir := os.Getenv("XDG_RUNTIME_DIR"); dir != "" {
         return filepath.Join(dir, "jarvis-worker-"+hash+".sock")
     }
     return fmt.Sprintf("/tmp/jarvis-%d-worker-%s.sock", os.Getuid(), hash)
 }

 Also add WorkerLogPath(hash string) string following the same LogDir() pattern.

 func (wm *WorkerManager) monitorWorker(info *WorkerInfo, cmd *exec.Cmd) {
     cmd.Wait()
     log.Printf("worker for %s (pid %d) exited", info.ModelPath, info.PID)
     wm.registry.Unregister(info.ModelPath)
     os.Remove(info.SocketPath)
 }

 func (wm *WorkerManager) KillWorker(info *WorkerInfo) error {
     // Send unload request to worker socket via client.ConnectTo(socketPath)
     // If no response within 10s, escalate to cmd.Process.Signal(syscall.SIGTERM)
 }

 func (wm *WorkerManager) ShutdownAll() {
     for _, w := range wm.registry.List() {
         wm.KillWorker(w)
     }
 }

 Step 8 — Refactor: daemon/handler.go

 Replace Handler struct. Keep ResponseWriter unchanged.

 // CoordinatorHandler replaces Handler in the main daemon.
 // Workers continue to use the existing Handler (renamed internally or kept as-is).
 type CoordinatorHandler struct {
     Registry      *Registry
     WorkerManager *WorkerManager
     Config        *config.Config
     StopCh        chan struct{}
 }

 handleLoad: Resolve alias → registry.RegisterIfAvailable (reserve) → wm.SpawnWorker → send OK. On any
 failure, unregister reservation.

 handleChat: Select worker by req.ModelPath (resolved) or auto-select (error if 0 or >1). Call
 proxyToWorker(worker.SocketPath, req, rw).

 proxyToWorker:
 func (h *CoordinatorHandler) proxyToWorker(socketPath string, req *protocol.Request, rw *ResponseWriter)
 error {
     conn, err := net.Dial("unix", socketPath)
     // ...
     data, _ := protocol.MarshalRequest(req)
     conn.Write(append(data, '\n'))
     scanner := bufio.NewScanner(conn)
     for scanner.Scan() {
         rw.w.Write(append(scanner.Bytes(), '\n')) // verbatim forwarding
         resp, _ := protocol.UnmarshalResponse(scanner.Bytes())
         if resp.Type == protocol.RespDone || resp.Type == protocol.RespError { break }
     }
 }

 handleUnload: Find worker by model path (or auto-select). Call wm.KillWorker(info).

 handleStatus: Build StatusPayload.Workers from registry.List(). For each worker, optionally query its socket
  for GPU mem stats. Populate legacy fields from Workers[0] for compat.

 handleStop: Call wm.ShutdownAll(), then signal StopCh.

 Verification

 1. jarvis start — daemon starts, no CUDA initialized in coordinator process
 2. jarvis load -g 0 -t 3600 /path/model-a.gguf — worker A spawned with CUDA_VISIBLE_DEVICES=0
 3. jarvis load -g 1 /path/model-b.gguf — worker B spawned with CUDA_VISIBLE_DEVICES=1
 4. jarvis load -g 0 /path/model-c.gguf — returns error: "GPU 0 already in use by model-a"
 5. jarvis "hello" — error: "multiple models loaded; use -m to select"
 6. jarvis -m /path/model-a.gguf "hello" — routes to worker A, streams response
 7. jarvis status — shows both workers with GPU IDs, PIDs, timeouts
 8. After 3600s — worker A's timeout fires, model A auto-unloads, GPU 0 lock released
 9. jarvis unload /path/model-b.gguf — sends unload to worker B, GPU 1 released
 10. jarvis stop — ShutdownAll() kills remaining workers, daemon exits
